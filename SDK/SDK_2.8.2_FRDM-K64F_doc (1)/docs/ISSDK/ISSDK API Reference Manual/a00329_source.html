40 /*! @brief Register settings for Normal Mode. */
41 const registerwritelist_t cDiffPConfigNormal[] = {
42  {DIFF_P_CTRL_REG2, DIFF_P_CTRL_REG2_ODR_ODR6P25, DIFF_P_CTRL_REG2_ODR_MASK},
43  {DIFF_P_CTRL_REG1, DIFF_P_CTRL_REG1_OSR_OSR512, DIFF_P_CTRL_REG1_OSR_MASK},

46 /*! @brief Register settings for Clearing Pressure and Temperature Data Ready Bits. */
47 const registerwritelist_t cDiffPClearStatusBits[] = {
48  {DIFF_P_INT_STATUS_0, 0x00, DIFF_P_INT_STATUS_0_TDR_MASK | DIFF_P_INT_STATUS_0_PDR_MASK}, __END_WRITE_DATA__};

50 /*! @brief Address of Status Register. */

53 /*! @brief Address and size of Raw Pressure+Temperature Data in Normal Mode. */
54 const registerreadlist_t cDiffPOutputNormal[] = {{.readFrom = DIFF_P_OUT_P_LSB, .numBytes = DIFF_P_DATA_SIZE},

60 /*!
61  * @brief Main function
62  */
63 int main(void)
64 {
65  int32_t status;
66  int8_t tempInDegrees;
67  float pressureInPascals;
68  diff_p_pressuredata_t rawData;
69  uint8_t whoAmI, conversionFactor;
70  uint8_t dataReady, data[DIFF_P_DATA_SIZE];
71 
72  diff_p_i2c_sensorhandle_t diff_pDriver;
73  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER;
74 
75  BOARD_InitPins();
76  BOARD_BootClockRUN();
77  BOARD_SystickEnable();
78  BOARD_InitDebugConsole();
79 
80  PRINTF("\r\n ISSDK DIFF-P Sensor Example !\r\n");
81 
82  /*! Initialize the I2C driver. */
83  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
84  if (ARM_DRIVER_OK != status)
85  {
86  PRINTF("\r\n I2C Initialization Failed.\r\n");
87  return -1;
88  }
89 
90  /*! Set the I2C Power mode. */
91  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
92  if (ARM_DRIVER_OK != status)
93  {
94  PRINTF("\r\n I2C Power Mode setting Failed.\r\n");
95  return -1;
96  }
97 
98  /*! Set the I2C bus speed. */
99  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
100  if (ARM_DRIVER_OK != status)
101  {
102  PRINTF("\r\n I2C Control Mode setting Failed.\r\n");
103  return -1;
104  }
105 
106  do
107  { /*! Initialize DIFF_P sensor driver. */
108  status = DIFF_P_I2C_Initialize(&diff_pDriver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, DIFF_P_I2C_ADDR,
109  DIFF_P_NPS3000VV_WHOAMI_VALUE);
110  if (SENSOR_ERROR_NONE == status)
111  {
112  whoAmI = DIFF_P_NPS3000VV_WHOAMI_VALUE;
113  conversionFactor = NPS3000VV_PRESSURE_DIV_FACTOR;
114  break;
115  }
116  status = DIFF_P_I2C_Initialize(&diff_pDriver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, DIFF_P_I2C_ADDR,
117  DIFF_P_NPS3001DV_WHOAMI_VALUE);
118  if (SENSOR_ERROR_NONE == status)
119  {
120  whoAmI = DIFF_P_NPS3001DV_WHOAMI_VALUE;
121  conversionFactor = NPS3001DV_PRESSURE_DIV_FACTOR;
122  break;
123  }
124  status = DIFF_P_I2C_Initialize(&diff_pDriver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, DIFF_P_I2C_ADDR,
125  DIFF_P_NPS3002VV_WHOAMI_VALUE);
126  if (SENSOR_ERROR_NONE == status)
127  {
128  whoAmI = DIFF_P_NPS3002VV_WHOAMI_VALUE;
129  conversionFactor = NPS3002VV_PRESSURE_DIV_FACTOR;
130  break;
131  }
132  status = DIFF_P_I2C_Initialize(&diff_pDriver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, DIFF_P_I2C_ADDR,
133  DIFF_P_NPS3005DV_WHOAMI_VALUE);
134  if (SENSOR_ERROR_NONE == status)
135  {
136  whoAmI = DIFF_P_NPS3005DV_WHOAMI_VALUE;
137  conversionFactor = NPS3005DV_PRESSURE_DIV_FACTOR;
138  break;
139  }
140 
141  PRINTF("\r\n Sensor Initialization Failed.\r\n");
142  return -1;
143  } while (false);
144  PRINTF("\r\n Initiliazed Sensor with WHO_AM_I = [0x%X].\r\n", whoAmI);
145 
146  /*! Set the task to be executed while waiting for I2C transactions to complete. */
147  DIFF_P_I2C_SetIdleTask(&diff_pDriver, (registeridlefunction_t)SMC_SetPowerModeVlpr, SMC);
148 
149  /*! Configure the DIFF_P sensor. */
150  status = DIFF_P_I2C_Configure(&diff_pDriver, cDiffPConfigNormal);
151  if (SENSOR_ERROR_NONE != status)
152  {
153  PRINTF("\r\nDIFF_P now active and entering data read loop...\r\n");
154  return -1;
155  }
156  PRINTF("\r\n Successfully Applied Sensor Poll Configuration.\r\n");
157 
158  for (;;) /* Forever loop */
159  {
160  /* Read INT_STATUS register */
161  status = DIFF_P_I2C_ReadData(&diff_pDriver, cDiffPStatus, &dataReady);
162  if (ARM_DRIVER_OK != status)
163  {
164  PRINTF("\r\n Read Failed. \r\n");
165  return -1;
166  }
167 
168  /*! Check for both data ready bits from the DIFF_P. */
169  if ((DIFF_P_INT_STATUS_0_PDR_DRDY | DIFF_P_INT_STATUS_0_TDR_DRDY) !=
170  (dataReady & (DIFF_P_INT_STATUS_0_PDR_MASK | DIFF_P_INT_STATUS_0_TDR_MASK)))
171  { /* Loop, if new samples are not available. */
172  continue;
173  }
174 
175  /*! Read new raw sensor data from the DIFF_P. */
176  status = DIFF_P_I2C_ReadData(&diff_pDriver, cDiffPOutputNormal, data);
177  if (ARM_DRIVER_OK != status)
178  {
179  PRINTF("\r\n Read Failed. \r\n");
180  return -1;
181  }
182 
183  /*! Explicitly clear the Status Bits. */
184  status = Sensor_I2C_Write(diff_pDriver.pCommDrv, &diff_pDriver.deviceInfo, diff_pDriver.slaveAddress,
185  cDiffPClearStatusBits);
186  if (ARM_DRIVER_OK != status)
187  {
188  PRINTF("\r\n Write Failed. \r\n");
189  return -1;
190  }
191 
192  /*! Process the sample and convert the raw sensor data. */
193  rawData.pressure = ((int16_t)(data[1]) << 8) | data[0];
194  pressureInPascals = (1.0 * rawData.pressure) / conversionFactor;
195 
196  rawData.temperature = (int8_t)(data[2]);
197  tempInDegrees = rawData.temperature;
198 
199  PRINTF("\r\n Pressure = %0.3f Pa", pressureInPascals);
200  PRINTF("\r\n Temperature = %d degC\r\n", tempInDegrees);
201  ASK_USER_TO_RESUME(50); /* Ask for user input after processing 50 samples. */
202  }
203 }
